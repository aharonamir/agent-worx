```python
# src/work/circuit_breaker.py

import os
import time
from datetime import datetime, timedelta
from src.core.models import CircuitBreakerStatus
from src.core.enums import CircuitState
from src.infra.redis_client import redis_get, redis_set, redis_hset, redis_hgetall

ERROR_RATE_THRESHOLD    = float(os.getenv("CIRCUIT_BREAKER_ERROR_RATE_THRESHOLD", "0.20"))
CONSECUTIVE_THRESHOLD   = int(os.getenv("CIRCUIT_BREAKER_CONSECUTIVE_FAILURES", "5"))
HALF_OPEN_PROBE_DELAY_S = 60   # seconds before allowing probe after OPEN
WINDOW_SECONDS          = 300  # 5-minute sliding window for error rate


def _state_key(agent_type: str) -> str:
    return f"circuit:{agent_type}:state"

def _stats_key(agent_type: str) -> str:
    return f"circuit:{agent_type}:stats"


def get_status(agent_type: str) -> CircuitBreakerStatus:
    stats = redis_hgetall(_stats_key(agent_type)) or {}
    state_val = redis_get(_state_key(agent_type)) or CircuitState.CLOSED.value

    error_rate          = float(stats.get("error_rate", "0.0"))
    consecutive         = int(stats.get("consecutive_failures", "0"))
    last_change_str     = stats.get("last_state_change")
    next_probe_str      = stats.get("next_probe_at")

    last_change  = datetime.fromisoformat(last_change_str) if last_change_str else datetime.utcnow()
    next_probe   = datetime.fromisoformat(next_probe_str)  if next_probe_str  else None

    return CircuitBreakerStatus(
        agent_type=agent_type,
        state=CircuitState(state_val),
        error_rate=error_rate,
        consecutive_failures=consecutive,
        last_state_change=last_change,
        next_probe_at=next_probe,
    )


def record_success(agent_type: str) -> None:
    """Record a successful task. May transition HALF_OPEN → CLOSED."""
    stats = redis_hgetall(_stats_key(agent_type)) or {}
    state = CircuitState(redis_get(_state_key(agent_type)) or CircuitState.CLOSED.value)

    consecutive = 0  # Reset on success
    _update_error_rate(agent_type, success=True)

    if state == CircuitState.HALF_OPEN:
        _transition(agent_type, CircuitState.CLOSED)

    redis_hset(_stats_key(agent_type), {
        "consecutive_failures": str(consecutive),
        "last_success": datetime.utcnow().isoformat(),
    })


def record_failure(agent_type: str) -> CircuitState:
    """Record a task failure. May open the circuit. Returns new state."""
    consecutive = int((redis_hgetall(_stats_key(agent_type)) or {}).get("consecutive_failures", "0")) + 1
    _update_error_rate(agent_type, success=False)

    redis_hset(_stats_key(agent_type), {"consecutive_failures": str(consecutive)})

    error_rate = float((redis_hgetall(_stats_key(agent_type)) or {}).get("error_rate", "0.0"))
    current_state = CircuitState(redis_get(_state_key(agent_type)) or CircuitState.CLOSED.value)

    should_open = (
        error_rate > ERROR_RATE_THRESHOLD or
        consecutive >= CONSECUTIVE_THRESHOLD
    )

    if should_open and current_state == CircuitState.CLOSED:
        probe_at = datetime.utcnow() + timedelta(seconds=HALF_OPEN_PROBE_DELAY_S)
        redis_hset(_stats_key(agent_type), {"next_probe_at": probe_at.isoformat()})
        _transition(agent_type, CircuitState.OPEN)
        return CircuitState.OPEN

    return current_state


def allow_request(agent_type: str) -> bool:
    """
    Return True if a task may proceed, False if it must route to human_queue.
    Invariant I9: OPEN circuit ALWAYS returns False.

    OPEN    → check if probe time has passed → if yes, transition to HALF_OPEN, allow one probe
    CLOSED  → True
    HALF_OPEN → True (one probe in flight)
    """
    status = get_status(agent_type)

    if status.state == CircuitState.CLOSED:
        return True

    if status.state == CircuitState.HALF_OPEN:
        return True

    if status.state == CircuitState.OPEN:
        if status.next_probe_at and datetime.utcnow() >= status.next_probe_at:
            _transition(agent_type, CircuitState.HALF_OPEN)
            return True
        return False  # Invariant I9: OPEN → False

    return False


def _transition(agent_type: str, new_state: CircuitState) -> None:
    redis_set(_state_key(agent_type), new_state.value)
    redis_hset(_stats_key(agent_type), {"last_state_change": datetime.utcnow().isoformat()})


def _update_error_rate(agent_type: str, success: bool) -> None:
    """Approximate sliding window error rate using exponential moving average."""
    stats = redis_hgetall(_stats_key(agent_type)) or {}
    current_rate = float(stats.get("error_rate", "0.0"))
    # EMA with alpha=0.1 (slow decay, reflects last ~10 tasks)
    alpha = 0.1
    new_rate = alpha * (0.0 if success else 1.0) + (1 - alpha) * current_rate
    redis_hset(_stats_key(agent_type), {"error_rate": str(round(new_rate, 4))})
```

```python
# src/temporal/workflows.py  — add to AgentTaskWorkflow before invoking LangGraph

from src.work.circuit_breaker import allow_request, record_success, record_failure
from src.core.models import EscalationEvent

@workflow.defn
class AgentTaskWorkflow:
    @workflow.run
    async def run(self, agent_type: str, task_input: dict) -> dict:

        # Invariant I9: check circuit breaker BEFORE any LLM call
        if not allow_request(agent_type):
            # Route to human queue immediately — no LangGraph invoke
            await workflow.execute_activity(
                route_to_human_queue,
                args=[agent_type, task_input, "circuit_open"],
                start_to_close_timeout=timedelta(seconds=30),
            )
            return {"status": "routed_to_human", "reason": "circuit_open"}

        try:
            result = await workflow.execute_activity(
                run_langgraph_activity,
                args=[agent_type, task_input],
                start_to_close_timeout=timedelta(minutes=30),
                retry_policy=RetryPolicy(maximum_attempts=3),
            )
            record_success(agent_type)
            return result
        except Exception as e:
            record_failure(agent_type)
            raise
```

## Acceptance Criteria

1. After 5 consecutive failures: `get_status().state == CircuitState.OPEN`
2. While OPEN: `allow_request()` returns `False`; subsequent task invocations go to human queue with no LLM call
3. After `HALF_OPEN_PROBE_DELAY_S` seconds: `allow_request()` returns `True` (HALF_OPEN probe)
4. Successful probe transitions HALF_OPEN → CLOSED
5. Failed probe transitions HALF_OPEN → OPEN (reset probe timer)

```python
# tests/phase_3/test_circuit_breaker.py

import pytest
from unittest.mock import patch
from src.work.circuit_breaker import (
    allow_request, record_success, record_failure,
    get_status, HALF_OPEN_PROBE_DELAY_S
)
from src.core.enums import CircuitState


@pytest.fixture(autouse=True)
def clear_redis(mock_redis):
    """Each test starts with clean Redis state."""
    mock_redis.flushall()


def test_initially_closed():
    status = get_status("test-agent")
    assert status.state == CircuitState.CLOSED
    assert allow_request("test-agent") is True


def test_opens_after_consecutive_failures():
    for _ in range(5):
        record_failure("test-agent")
    status = get_status("test-agent")
    assert status.state == CircuitState.OPEN


def test_open_circuit_routes_to_human_queue():
    """Invariant I9: OPEN circuit must return False from allow_request."""
    for _ in range(5):
        record_failure("test-agent")
    assert allow_request("test-agent") is False


def test_success_resets_consecutive_failures():
    for _ in range(3):
        record_failure("test-agent")
    record_success("test-agent")
    status = get_status("test-agent")
    assert status.consecutive_failures == 0


def test_half_open_probe_after_delay(freezer):
    """After delay, OPEN transitions to HALF_OPEN and allows one request."""
    for _ in range(5):
        record_failure("test-agent")
    assert allow_request("test-agent") is False

    # Advance time past probe delay
    freezer.move_to(HALF_OPEN_PROBE_DELAY_S + 1)
    assert allow_request("test-agent") is True
    assert get_status("test-agent").state == CircuitState.HALF_OPEN


def test_successful_probe_closes_circuit(freezer):
    for _ in range(5):
        record_failure("test-agent")
    freezer.move_to(HALF_OPEN_PROBE_DELAY_S + 1)
    allow_request("test-agent")          # transitions to HALF_OPEN
    record_success("test-agent")         # successful probe
    assert get_status("test-agent").state == CircuitState.CLOSED
```
