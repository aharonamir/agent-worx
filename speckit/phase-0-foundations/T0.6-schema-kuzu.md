## What to Build

The schema must be applied once per agent type at creation time (called from T1.1).
It defines both the task namespace and the coordination namespace.

```python
# Add to src/infra/kuzu_client.py

SCHEMA_DDL = [
    # ── Task namespace ──────────────────────────────────────────────
    """CREATE NODE TABLE IF NOT EXISTS Concept(
        id STRING,
        label STRING,
        namespace STRING,        -- 'task' or 'coordination'
        confidence FLOAT,
        created_at STRING,
        PRIMARY KEY(id)
    )""",
    """CREATE NODE TABLE IF NOT EXISTS Constraint(
        id STRING,
        description STRING,
        severity STRING,         -- 'hard' | 'soft'
        PRIMARY KEY(id)
    )""",
    """CREATE REL TABLE IF NOT EXISTS RELATES_TO(
        FROM Concept TO Concept,
        relation STRING,
        confidence FLOAT,
        version INT
    )""",
    """CREATE REL TABLE IF NOT EXISTS HAS_CONSTRAINT(
        FROM Concept TO Constraint
    )""",

    # ── Coordination namespace ──────────────────────────────────────
    """CREATE NODE TABLE IF NOT EXISTS AgentRole(
        id STRING,
        name STRING,
        input_schema STRING,     -- JSON string of {field: type}
        output_schema STRING,    -- JSON string of {field: type}
        prohibited_fields STRING, -- JSON array of strings
        PRIMARY KEY(id)
    )""",
    """CREATE REL TABLE IF NOT EXISTS HANDS_OFF_TO(
        FROM AgentRole TO AgentRole,
        condition STRING,
        validates STRING,        -- JSON array of field names to assert
        confidence FLOAT
    )""",
    """CREATE REL TABLE IF NOT EXISTS HAS_ROLE(
        FROM Concept TO AgentRole
    )""",
]

def initialize_schema(agent_type: str) -> None:
    """Apply the domain graph schema to a new agent type's Kuzu DB.
    Safe to call multiple times — all statements use IF NOT EXISTS.
    """
    conn = get_connection(agent_type)
    for ddl in SCHEMA_DDL:
        conn.execute(ddl)
```

## Acceptance Criteria

1. `initialize_schema("test-agent")` completes without error
2. All 4 node tables exist after initialization: `Concept`, `Constraint`, `AgentRole` — verify with a test insert + query on each
3. All 3 rel tables exist: `RELATES_TO`, `HAS_CONSTRAINT`, `HANDS_OFF_TO`
4. Calling `initialize_schema` twice on the same agent type does not raise an error

## Test

```python
# tests/phase_0/test_schemas.py

def test_kuzu_schema_initializes(tmp_path, monkeypatch):
    monkeypatch.setenv("KUZU_GRAPHS_DIR", str(tmp_path))
    from src.infra.kuzu_client import initialize_schema, execute

    initialize_schema("test-agent")

    # Insert and query a Concept node
    execute("test-agent",
        "CREATE (:Concept {id: 'c1', label: 'burger', namespace: 'task', confidence: 0.9, created_at: '2025-01-01'})"
    )
    result = execute("test-agent", "MATCH (c:Concept) RETURN c.label").get_as_df()
    assert "burger" in result["c.label"].values

    # Insert and query a RELATES_TO edge
    execute("test-agent",
        "CREATE (:Concept {id: 'c2', label: 'menu', namespace: 'task', confidence: 0.8, created_at: '2025-01-01'})"
    )
    execute("test-agent",
        "MATCH (a:Concept {id:'c1'}),(b:Concept {id:'c2'}) CREATE (a)-[:RELATES_TO {relation:'has', confidence:0.85, version:1}]->(b)"
    )
    result = execute("test-agent",
        "MATCH (a:Concept)-[r:RELATES_TO]->(b:Concept) RETURN a.label, r.relation, b.label"
    ).get_as_df()
    assert len(result) == 1
    assert result.iloc[0]["r.relation"] == "has"

def test_kuzu_schema_idempotent(tmp_path, monkeypatch):
```
