## What to Build

A thin connection manager that gives the rest of the codebase a Kuzu connection
scoped to a specific agent type. Enforces Invariant I10 (one DB per type, not per instance).

```python
# src/infra/kuzu_client.py

import os
from pathlib import Path
import kuzu

KUZU_GRAPHS_DIR = Path(os.getenv("KUZU_GRAPHS_DIR", "./graphs"))

_connections: dict[str, kuzu.Connection] = {}

def get_db(agent_type: str) -> kuzu.Database:
    """Return the shared Kuzu Database for this agent type.
    Creates the directory and DB file if they do not exist.
    """
    db_path = KUZU_GRAPHS_DIR / agent_type / "domain.kuzu"
    db_path.parent.mkdir(parents=True, exist_ok=True)
    return kuzu.Database(str(db_path))

def get_connection(agent_type: str) -> kuzu.Connection:
    """Return a cached Connection for this agent type.
    INVARIANT I10: path is always graphs/{agent_type}/domain.kuzu
    Never graphs/{agent_type}/{instance_id}/domain.kuzu
    """
    if agent_type not in _connections:
        _connections[agent_type] = kuzu.Connection(get_db(agent_type))
    return _connections[agent_type]

def execute(agent_type: str, query: str, params: dict | None = None):
    """Execute a Cypher query and return results as a DataFrame."""
    conn = get_connection(agent_type)
    if params:
        return conn.execute(query, params)
    return conn.execute(query)
```

## Acceptance Criteria

1. `get_connection("burger-order")` creates file at `./graphs/burger-order/domain.kuzu` (not at any instance-level path)
2. Two calls to `get_connection("burger-order")` return the same connection object
3. Multi-hop path query on a 100K node test graph completes in < 50ms

```python
# tests/phase_0/test_infra_connectivity.py

def test_kuzu_shared_per_type_not_per_instance(tmp_path, monkeypatch):
    """Invariant I10: DB path must be per agent type, never per instance."""
    monkeypatch.setenv("KUZU_GRAPHS_DIR", str(tmp_path))
    from src.infra.kuzu_client import get_connection, get_db
    import kuzu

    db_path = tmp_path / "burger-order" / "domain.kuzu"
    conn = get_connection("burger-order")
    assert db_path.exists(), f"Expected DB at {db_path}"
    assert isinstance(conn, kuzu.Connection)

    # Same agent type returns same connection
    conn2 = get_connection("burger-order")
    assert conn is conn2

    # Different agent type gets different DB
    conn3 = get_connection("payment-agent")
    assert conn is not conn3

def test_kuzu_multihop_query_performance(tmp_path, monkeypatch):
    """Multi-hop path query must complete < 50ms on a seeded graph."""
    import time
    monkeypatch.setenv("KUZU_GRAPHS_DIR", str(tmp_path))
    from src.infra.kuzu_client import get_connection, execute
    from phase_0_fixtures import seed_kuzu_test_graph  # helper that inserts 100K nodes

    seed_kuzu_test_graph("perf-test-agent")
    start = time.perf_counter()
    result = execute(
        "perf-test-agent",
        "MATCH (a:Concept)-[:RELATES_TO*1..3]->(b:Concept) WHERE a.label = $label RETURN b LIMIT 20",
        {"label": "root-concept"}
    )
    elapsed_ms = (time.perf_counter() - start) * 1000
    assert elapsed_ms < 50, f"Query took {elapsed_ms:.1f}ms, expected < 50ms"
```

## Notes
- Install Vela Engineering fork for multi-writer support:
  `pip install git+https://github.com/Vela-Engineering/kuzu`
  Standard `kuzu` from PyPI is single-writer and will deadlock under concurrent agent activity.
- The `_connections` dict is process-scoped. In a multi-process deployment each worker process
  maintains its own connection cache. This is correct — Kuzu handles concurrent reads safely.
