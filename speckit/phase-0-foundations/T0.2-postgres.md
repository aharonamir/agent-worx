## What to Build

Three logical databases on one Postgres instance:

| Database | Purpose | Owner |
|---|---|---|
| `agent_checkpoints` | LangGraph `AsyncPostgresSaver` state | LangGraph (T2.1) |
| `agent_knowledge` | `agent_types`, `delta_entries`, `team_compositions`, `cert_events`, `shadow_reviews`, `escalation_events` | Apprentice/Journeyman/Work app code |
| `agent_ops` | Temporal's internal persistence | Temporal cluster |

`agent_checkpoints` is the default DB created by the `postgres` image on first boot.
`init.sql` creates the other two and installs extensions.

### Connection Pool Client

```python
# src/infra/postgres_client.py

import os
import asyncpg
from typing import Optional

_pool: Optional[asyncpg.Pool] = None

KNOWLEDGE_DB_URL = os.environ["KNOWLEDGE_DB_URL"]   # postgresql://...@postgres:5432/agent_knowledge


async def init_pool() -> asyncpg.Pool:
    """Create the connection pool. Call once at app startup."""
    global _pool
    if _pool is None:
        # asyncpg uses postgresql:// not postgresql+asyncpg://
        dsn = KNOWLEDGE_DB_URL.replace("postgresql+asyncpg://", "postgresql://")
        _pool = await asyncpg.create_pool(
            dsn=dsn,
            min_size=2,
            max_size=20,
            command_timeout=30,
        )
    return _pool


def get_pool() -> asyncpg.Pool:
    """Return the existing pool. Raises if init_pool() was not called."""
    if _pool is None:
        raise RuntimeError("Postgres pool not initialized — call init_pool() at startup")
    return _pool


async def close_pool() -> None:
    global _pool
    if _pool is not None:
        await _pool.close()
        _pool = None
```

```python
# alembic/env.py

import os
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context

config = context.config
fileConfig(config.config_file_name)

# Override URL from env var — use sync driver for Alembic
db_url = os.environ["KNOWLEDGE_DB_URL"].replace("postgresql+asyncpg://", "postgresql+psycopg://")
config.set_main_option("sqlalchemy.url", db_url)

target_metadata = None  # raw SQL migrations, no ORM models


def run_migrations_offline():
    context.configure(url=db_url, target_metadata=target_metadata, literal_binds=True)
    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online():
    connectable = engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(connection=connection, target_metadata=target_metadata)
        with context.begin_transaction():
            context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

## Acceptance Criteria

1. `docker compose up -d postgres` brings the container to `healthy` within 15 seconds
2. `psql -h localhost -U alf -d agent_checkpoints -c '\l'` lists all three databases: `agent_checkpoints`, `agent_knowledge`, `agent_ops`
3. `agent_knowledge` has both `uuid-ossp` and `vector` extensions installed
4. `alembic upgrade head` runs without error and creates the `agent_types` table in `agent_knowledge`
5. `init_pool()` followed by `get_pool()` returns a working pool; a test query (`SELECT 1`) succeeds
6. Connection pool survives 50 concurrent acquisitions without exhaustion (`max_size=20` with proper release)

@pytest.mark.asyncio
async def test_connection_pool_basic_query(monkeypatch):
    monkeypatch.setenv("KNOWLEDGE_DB_URL", "postgresql+asyncpg://alf:alfpass@localhost:5432/agent_knowledge")

    pool = get_pool()
    async with pool.acquire() as conn:
        result = await conn.fetchval("SELECT 1")
    assert result == 1

@pytest.mark.asyncio
async def test_pool_handles_concurrent_acquisitions(monkeypatch):
    monkeypatch.setenv("KNOWLEDGE_DB_URL", "postgresql+asyncpg://alf:alfpass@localhost:5432/agent_knowledge")
    from src.infra.postgres_client import init_pool, get_pool, close_pool

    await init_pool()
    pool = get_pool()

    async def query():
        async with pool.acquire() as conn:
            await asyncio.sleep(0.05)
            return await conn.fetchval("SELECT 1")

    results = await asyncio.gather(*[query() for _ in range(50)])
    assert all(r == 1 for r in results)
    await close_pool()


def test_agent_types_table_exists_after_migration():

        return row

    assert row["data_type"] == "character varying"

## Notes

- **Why three separate databases instead of one with schemas:** Temporal's auto-setup expects
  a dedicated database it fully manages (`agent_ops`). Mixing Temporal's internal tables with
  application tables in the same DB makes upgrades and backups riskier. Three databases on one
  Postgres instance is the right isolation level without the operational cost of three instances.
- **`agent_checkpoints` is the default DB** created by the `postgres:16.4` image via
  `POSTGRES_DB` env var. `init.sql` only needs to create the other two.
- T0.7 adds the remaining tables (`delta_entries`, `team_compositions`, `cert_events`,
  `shadow_reviews`, `escalation_events`) as migration `002`. Run migrations in order:
  `001_initial_schema` then `002_cert_events` (per `shared/PROJECT_STRUCTURE.md`).
