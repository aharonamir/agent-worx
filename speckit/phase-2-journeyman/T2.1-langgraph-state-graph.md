```python
# src/journeyman/graph_definition.py

import os
from typing import Literal
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.postgres.aio import AsyncPostgresSaver
from src.core.models import AgentState
from src.journeyman.retrieval import retrieve_from_certified_kb, retrieve_handoff_from_graph
from src.work.scope_detector import compute_scope_score
from src.work.manifest_enforcer import enforce_action

# NEVER import SqliteSaver — see INVARIANTS.md I4
# from langgraph.checkpoint.sqlite import SqliteSaver  ← FORBIDDEN

DATABASE_URL  = os.environ["DATABASE_URL"]
MAX_ITERATIONS = int(os.getenv("MAX_AGENT_ITERATIONS", "10"))


async def build_journeyman_graph(agent_type: str) -> StateGraph:
    """
    Build the LangGraph StateGraph for a journeyman-phase agent.

    Nodes:
        retrieve   — load relevant knowledge from certified KB + domain graph
        execute    — run the agent's core task logic
        validate   — check output quality and scope
        route      — decide next step (Literal routing, no string matching)

    Invariant I5: every loop must check iterations < max_iterations.
    """
    graph = StateGraph(AgentState)

    graph.add_node("retrieve", _retrieve_node)
    graph.add_node("execute",  _execute_node)
    graph.add_node("validate", _validate_node)
    graph.add_node("escalate", _escalate_node)

    graph.set_entry_point("retrieve")

    graph.add_edge("retrieve", "execute")

    # Conditional routing from validate — uses Literal type, never string matching
    graph.add_conditional_edges(
        "validate",
        _route_after_validate,
        {
            "complete":  END,
            "escalate":  "escalate",
            "handoff":   END,          # orchestrator handles handoff
            "continue":  "retrieve",   # loop back for multi-step tasks
        }
    )
    graph.add_edge("execute", "validate")
    graph.add_edge("escalate", END)

    checkpointer = AsyncPostgresSaver.from_conn_string(DATABASE_URL)
    return graph.compile(checkpointer=checkpointer)


# ── Node implementations ──────────────────────────────────────────────────────

async def _retrieve_node(state: AgentState) -> AgentState:
    """Load relevant context from certified KB (Qdrant) and domain graph (Kuzu)."""
    kb_results = await retrieve_from_certified_kb(
        agent_type=state.agent_type,
        query=str(state.task_input),
        top_k=5,
    )
    handoff_rules = retrieve_handoff_from_graph(
        agent_type=state.agent_type,
        current_role=state.agent_type,
    )
    state.action_log.append({
        "step": "retrieve",
        "kb_results_count": len(kb_results),
        "handoff_rules_count": len(handoff_rules),
    })
    return state


async def _execute_node(state: AgentState) -> AgentState:
    """Core task execution. Implement domain-specific logic here."""
    # Increment iterations FIRST — guard checked in validate
    state.iterations += 1
    state.action_log.append({"step": "execute", "iteration": state.iterations})
    # TODO: inject LLM call with task_input + retrieved context
    return state


async def _validate_node(state: AgentState) -> AgentState:
    """Validate output quality. Set routing_decision for conditional edge."""
    # Invariant I5: must check iterations ceiling
    if state.iterations >= state.max_iterations:
        state.escalation_required = True
        state.escalation_reason = "max_iterations"
        state.routing_decision = "escalate"
        return state

    if state.escalation_required:
        state.routing_decision = "escalate"
        return state

    # Determine if task is complete or needs another loop
    # Default to complete — override in domain-specific subclass
    state.routing_decision = "complete"
    return state


async def _escalate_node(state: AgentState) -> AgentState:
    """Write escalation event. Workers escalate to orchestrator, not directly to humans."""
    from src.infra.postgres_client import insert_escalation_event
    from src.core.models import EscalationEvent
    from src.core.enums import Phase

    # Invariant I7: workers route to orchestrator, not human_queue
    # Orchestrator agents (topology=orchestrated) may route to human_queue
    routed_to = "human_queue"  # override to "orchestrator" for worker agents

    event = EscalationEvent(
        task_id=state.task_id,
        agent_type=state.agent_type,
        instance_id=state.instance_id,
        reason=state.escalation_reason or "unknown",
        task_summary=str(state.task_input)[:500],
        routed_to=routed_to,
    )
    await insert_escalation_event(event)
    state.action_log.append({"step": "escalate", "reason": state.escalation_reason})
    return state


def _route_after_validate(state: AgentState) -> Literal["complete", "escalate", "handoff", "continue"]:
    """
    Routing function for conditional edges.
    MUST return a Literal type — never a dynamic string.
    """
    return state.routing_decision  # type: ignore[return-value]
```

## Acceptance Criteria

1. `build_journeyman_graph()` compiles without error; `AsyncPostgresSaver` is the checkpointer (not SqliteSaver)
2. Agent reaching `max_iterations` exits via `escalate` node with `escalation_reason="max_iterations"`
3. `_route_after_validate` returns only valid Literal values — no other string values
4. A full run with `task_input={}` completes without error and logs `retrieve` + `execute` + `validate` steps
