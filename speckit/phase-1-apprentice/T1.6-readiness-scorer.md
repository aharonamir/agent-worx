```python
# src/apprentice/readiness_scorer.py

import os
from datetime import datetime
from src.core.models import ReadinessScore
from src.infra.kuzu_client import execute
from src.infra.redis_client import redis_set
from src.core.redis_keys import (
    agent_cert_ready_key,
    agent_proposal_ready_key,
    agent_readiness_score_key,
)

PROPOSAL_THRESHOLD = float(os.getenv("READINESS_PROPOSAL_THRESHOLD", "0.55"))
CERT1_THRESHOLD    = float(os.getenv("READINESS_CERT1_THRESHOLD", "0.80"))
EDGE_DENSITY_CAP   = 5.0   # avg edges per node considered "saturated"


def compute_and_store(agent_type: str) -> ReadinessScore:
    """
    Compute the readiness score, persist to Redis, and return the result.

    Formula:
        score = (0.4 × node_coverage) + (0.35 × edge_density) + (0.25 × confidence_mean)

    node_coverage  = nodes_with_at_least_one_edge / total_seed_topic_nodes
    edge_density   = min(avg_edges_per_node / EDGE_DENSITY_CAP, 1.0)
    confidence_mean = mean(confidence) across all RELATES_TO edges (0.0 if no edges)
    """
    node_coverage   = _compute_node_coverage(agent_type)
    edge_density    = _compute_edge_density(agent_type)
    confidence_mean = _compute_confidence_mean(agent_type)

    score = round(
        (0.40 * node_coverage) +
        (0.35 * edge_density)  +
        (0.25 * confidence_mean),
        4
    )

    result = ReadinessScore(
        agent_type_id=agent_type,
        score=score,
        node_coverage=round(node_coverage, 4),
        edge_density=round(edge_density, 4),
        confidence_mean=round(confidence_mean, 4),
        proposal_ready=score >= PROPOSAL_THRESHOLD,
        cert_ready=score >= CERT1_THRESHOLD,
        computed_at=datetime.utcnow(),
    )

    # Persist to Redis
    redis_set(agent_readiness_score_key(agent_type), str(score))
    redis_set(agent_proposal_ready_key(agent_type), str(result.proposal_ready).lower())
    redis_set(agent_cert_ready_key(agent_type), str(result.cert_ready).lower())

    return result


def _compute_node_coverage(agent_type: str) -> float:
    df = execute(agent_type, """
        MATCH (c:Concept WHERE c.namespace = 'task')
        OPTIONAL MATCH (c)-[r:RELATES_TO]-()
        WITH COUNT(DISTINCT c) AS total,
             COUNT(DISTINCT CASE WHEN r IS NOT NULL THEN c END) AS covered
        RETURN total, covered
    """).get_as_df()
    total   = int(df["total"].iloc[0])
    covered = int(df["covered"].iloc[0])
    return covered / total if total > 0 else 0.0


def _compute_edge_density(agent_type: str) -> float:
    df = execute(agent_type, """
        MATCH (c:Concept)
        OPTIONAL MATCH (c)-[r:RELATES_TO]->()
        WITH COUNT(DISTINCT c) AS node_count, COUNT(r) AS edge_count
        RETURN node_count, edge_count
    """).get_as_df()
    nodes = int(df["node_count"].iloc[0])
    edges = int(df["edge_count"].iloc[0])
    if nodes == 0:
        return 0.0
    avg = edges / nodes
    return min(avg / EDGE_DENSITY_CAP, 1.0)


def _compute_confidence_mean(agent_type: str) -> float:
    df = execute(agent_type, """
        MATCH ()-[r:RELATES_TO]->()
        RETURN AVG(r.confidence) AS mean_conf
    """).get_as_df()
    val = df["mean_conf"].iloc[0]
    return float(val) if val is not None else 0.0
```

## Acceptance Criteria

1. Empty graph → `score == 0.0`, `proposal_ready == False`, `cert_ready == False`
2. Fully connected graph with all edges at confidence 1.0 → `score == 1.0`
3. Score increases monotonically as nodes and edges are added (tested over 5 incremental graph states)
4. `proposal_ready == True` when score ≥ `READINESS_PROPOSAL_THRESHOLD` (default 0.55)
5. `cert_ready == True` when score ≥ `READINESS_CERT1_THRESHOLD` (default 0.80)
6. Redis keys `agent:{type}:readiness_score` and `agent:{type}:proposal_ready` are updated after each call

```python
# tests/phase_1/test_readiness_scorer.py

import pytest
from src.apprentice.readiness_scorer import compute_and_store
from src.infra.kuzu_client import initialize_schema, execute
from src.infra.redis_client import redis_get


@pytest.fixture
def agent(tmp_path, monkeypatch):
    monkeypatch.setenv("KUZU_GRAPHS_DIR", str(tmp_path))
    monkeypatch.setenv("READINESS_PROPOSAL_THRESHOLD", "0.55")
    monkeypatch.setenv("READINESS_CERT1_THRESHOLD", "0.80")
    atype = "scorer-test"
    initialize_schema(atype)
    return atype


def test_empty_graph_scores_zero(agent):
    result = compute_and_store(agent)
    assert result.score == 0.0
    assert result.proposal_ready is False
    assert result.cert_ready is False


def test_score_increases_monotonically(agent):
    scores = []
    for i in range(5):
        execute(agent,
            f"CREATE (:Concept {{id:'c{i}', label:'concept{i}', namespace:'task', confidence:0.9, created_at:'2025-01-01'}})"
        )
        if i > 0:
            execute(agent,
                f"MATCH (a:Concept{{id:'c{i-1}'}}),(b:Concept{{id:'c{i}'}}) "
                f"CREATE (a)-[:RELATES_TO{{relation:'relates',confidence:0.85,version:1}}]->(b)"
            )
        result = compute_and_store(agent)
        scores.append(result.score)

    for j in range(1, len(scores)):
        assert scores[j] >= scores[j-1], f"Score decreased at step {j}: {scores}"


def test_proposal_ready_at_threshold(agent, monkeypatch):
    monkeypatch.setenv("READINESS_PROPOSAL_THRESHOLD", "0.0")  # Force immediate
    execute(agent,
        "CREATE (:Concept {id:'c1', label:'burger', namespace:'task', confidence:0.9, created_at:'2025-01-01'})"
    )
    result = compute_and_store(agent)
    assert result.proposal_ready is True


def test_redis_updated_after_compute(agent):
    execute(agent,
        "CREATE (:Concept {id:'c1', label:'test', namespace:'task', confidence:0.9, created_at:'2025-01-01'})"
    )
    result = compute_and_store(agent)
    from src.core.redis_keys import agent_readiness_score_key

    stored = redis_get(agent_readiness_score_key(agent))
    assert stored is not None
    assert abs(float(stored) - result.score) < 0.0001


def test_formula_correctness():
    """Unit test the formula in isolation with known inputs."""
    from src.apprentice.readiness_scorer import EDGE_DENSITY_CAP
    node_coverage   = 0.8
    avg_edges       = 3.0
    edge_density    = min(avg_edges / EDGE_DENSITY_CAP, 1.0)  # 0.6
    confidence_mean = 0.9
    expected = round((0.40 * 0.8) + (0.35 * 0.6) + (0.25 * 0.9), 4)
    assert expected == pytest.approx(0.755, abs=0.001)
```
