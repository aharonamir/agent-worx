## What to Build

This is the entry point for the entire framework. Creating an agent type does four things atomically:
1. Writes the agent type record to Postgres (`agent_types` table)
2. Initializes the Kuzu domain graph for this agent type (calls T0.6's `initialize_schema`)
3. Writes root concept nodes to Kuzu — one per `topic_list` entry, namespace="task"
4. Sets initial operational state in Redis: `phase=apprentice`, `learning_rate=1.0`

If any step fails, the whole operation must roll back — no partial agent type should exist.

### Postgres Table

```sql
-- Add to alembic/versions/001_initial_schema.py

CREATE TABLE agent_types (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name            VARCHAR(100) NOT NULL UNIQUE,
    goal            TEXT NOT NULL,
    task_boundary   TEXT NOT NULL,
    topic_list      JSONB NOT NULL,
    topology_type   VARCHAR(20) NOT NULL,
    workflow_participants JSONB NOT NULL DEFAULT '[]',
    created_at      TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE UNIQUE INDEX idx_agent_types_name ON agent_types(name);
```

### Service Layer

```python
# src/core/agent_type_service.py

import json
import uuid
from datetime import datetime
from src.core.models import AgentTypeCreate, AgentTypeResponse
from src.core.enums import Phase
from src.core.exceptions import AgentTypeAlreadyExists, AgentTypeNotFound
from src.infra.postgres_client import get_pool
from src.infra.kuzu_client import initialize_schema, execute as kuzu_execute
from src.infra.redis_client import redis_delete, redis_get, redis_set
from src.core.redis_keys import (
    agent_learning_rate_key,
    agent_phase_key,
    agent_readiness_score_key,
)


async def create_agent_type(payload: AgentTypeCreate) -> AgentTypeResponse:
    """
    Create a new agent type. Atomic across Postgres + Kuzu + Redis.

    Steps:
      1. Insert into agent_types table (Postgres) -- fails on duplicate name
      2. Initialize Kuzu schema for this agent type
      3. Write root Concept nodes for each topic_list entry
      4. Initialize Redis operational state (phase, learning_rate, readiness)

    Rollback: if step 2-4 fails after step 1 succeeds, delete the Postgres
    row, remove any Redis keys that were written, and delete root Concept
    nodes created during this call -- avoids orphaned cross-store state.
    """
    pool = get_pool()
    agent_type_id = str(uuid.uuid4())

    # Step 1: Postgres insert
    async with pool.acquire() as conn:
        existing = await conn.fetchrow(
            "SELECT id FROM agent_types WHERE name = $1", payload.name
        )
        if existing:
            raise AgentTypeAlreadyExists(payload.name)

        await conn.execute(
            """
            INSERT INTO agent_types
                (id, name, goal, task_boundary, topic_list, topology_type, workflow_participants, created_at)
            VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
            """,
            agent_type_id, payload.name, payload.goal, payload.task_boundary,
            json.dumps(payload.topic_list), payload.topology_type.value,
            json.dumps(payload.workflow_participants), datetime.utcnow(),
        )

    created_concept_ids: list[str] = []
    redis_keys = [
        agent_phase_key(payload.name),
        agent_learning_rate_key(payload.name),
        agent_readiness_score_key(payload.name),
    ]

    try:
        # Step 2: Initialize Kuzu schema (idempotent -- safe even if called twice)
        # NOTE: agent_type identifier used for Kuzu path is the NAME, not the UUID --
        # this keeps ./graphs/{name}/domain.kuzu human-readable and matches
        # Invariant I10 (one Kuzu DB per agent TYPE name).
        initialize_schema(payload.name)

        # Step 3: Write root concept nodes -- one per topic_list entry
        now_str = datetime.utcnow().isoformat()
        for i, topic in enumerate(payload.topic_list):
            concept_id = f"root-{i}-{uuid.uuid4().hex[:8]}"
            kuzu_execute(payload.name, """
                CREATE (:Concept {
                    id: $id, label: $label, namespace: 'task',
                    confidence: 0.0, created_at: $created_at
                })
            """, {"id": concept_id, "label": topic, "created_at": now_str})
            created_concept_ids.append(concept_id)

        # Step 4: Redis operational state
        redis_set(agent_phase_key(payload.name), Phase.APPRENTICE.value)
        redis_set(agent_learning_rate_key(payload.name), "1.0")
        redis_set(agent_readiness_score_key(payload.name), "0.0")

    except Exception:
        # Roll back all cross-store state created by this operation.
        for key in redis_keys:
            redis_delete(key)
        for concept_id in created_concept_ids:
            kuzu_execute(payload.name, "MATCH (c:Concept {id: $id}) DELETE c", {"id": concept_id})
        async with pool.acquire() as conn:
            await conn.execute("DELETE FROM agent_types WHERE id = $1", agent_type_id)
        raise

    return AgentTypeResponse(
        agent_type_id=agent_type_id,
        name=payload.name,
        goal=payload.goal,
        task_boundary=payload.task_boundary,
        topic_list=payload.topic_list,
        topology_type=payload.topology_type,
        workflow_participants=payload.workflow_participants,
        phase=Phase.APPRENTICE,
        learning_rate=1.0,
        readiness_score=0.0,
        graph_initialized=True,
        created_at=datetime.utcnow(),
    )


async def get_agent_type(agent_type_id: str) -> AgentTypeResponse:
    """Fetch an agent type by ID, merging Postgres record with live Redis state."""
    pool = get_pool()
    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            "SELECT * FROM agent_types WHERE id = $1", agent_type_id
        )
    if row is None:
        raise AgentTypeNotFound(agent_type_id)

    from src.core.enums import TopologyType

    phase = redis_get(agent_phase_key(row["name"])) or Phase.APPRENTICE.value
    learning_rate = float(redis_get(agent_learning_rate_key(row["name"])) or "1.0")
    readiness = float(redis_get(agent_readiness_score_key(row["name"])) or "0.0")

    return AgentTypeResponse(
        agent_type_id=str(row["id"]),
        name=row["name"],
        goal=row["goal"],
        task_boundary=row["task_boundary"],
        topic_list=json.loads(row["topic_list"]) if isinstance(row["topic_list"], str) else row["topic_list"],
        topology_type=TopologyType(row["topology_type"]),
        workflow_participants=json.loads(row["workflow_participants"]) if isinstance(row["workflow_participants"], str) else row["workflow_participants"],
        phase=Phase(phase),
        learning_rate=learning_rate,
        readiness_score=readiness,
        graph_initialized=True,
        created_at=row["created_at"],
    )
```

class AgentTypeAlreadyExists(Exception):
    def __init__(self, name: str):
        self.name = name
        super().__init__(f"Agent type '{name}' already exists")

class AgentTypeNotFound(Exception):
    def __init__(self, agent_type_id: str):
        self.agent_type_id = agent_type_id
        super().__init__(f"Agent type '{agent_type_id}' not found")

### API Route

```python
# src/api/routes/agent_types.py

from fastapi import APIRouter, HTTPException, status
from src.core.models import AgentTypeCreate, AgentTypeResponse
from src.core.exceptions import AgentTypeAlreadyExists, AgentTypeNotFound
from src.core.agent_type_service import create_agent_type, get_agent_type

router = APIRouter(prefix="/agent-types", tags=["agent-types"])


@router.post("", response_model=AgentTypeResponse, status_code=status.HTTP_201_CREATED)
async def create(payload: AgentTypeCreate) -> AgentTypeResponse:
    if len(payload.topic_list) == 0:
        raise HTTPException(status_code=400, detail="topic_list must not be empty")
    try:
        return await create_agent_type(payload)
    except AgentTypeAlreadyExists as e:
        raise HTTPException(status_code=409, detail=str(e))


@router.get("/{agent_type_id}", response_model=AgentTypeResponse)
async def get(agent_type_id: str) -> AgentTypeResponse:
    try:
        return await get_agent_type(agent_type_id)
    except AgentTypeNotFound as e:
        raise HTTPException(status_code=404, detail=str(e))
```

## Acceptance Criteria

1. `POST /api/v1/agent-types` with a valid payload returns `201` and `graph_initialized: true`
2. The Kuzu graph at `./graphs/{name}/domain.kuzu` exists after creation, with one `Concept` node per `topic_list` entry, all with `namespace='task'` and `confidence=0.0`
3. Redis keys `agent:{name}:phase`, `agent:{name}:learning_rate`, `agent:{name}:readiness_score` are set to `apprentice`, `1.0`, `0.0` respectively
4. `POST /api/v1/agent-types` with a duplicate `name` returns `409`
5. `POST /api/v1/agent-types` with empty `topic_list` returns `400`
6. `GET /api/v1/agent-types/{id}` for a non-existent ID returns `404`
7. If Kuzu initialization fails (simulate with a monkeypatch), the Postgres `agent_types` row is rolled back -- no orphaned record remains

## Notes

- **Agent type identifier choice:** Kuzu graph paths and Redis keys use the agent type **name** (e.g. `burger-order`), not the UUID `agent_type_id`. The Postgres `id` (UUID) is the canonical primary key returned to API clients, but the `name` is the cross-store identifier because it's human-readable and stable. Both `T1.2`-`T1.11` and Phase 2/3 tasks reference agent types by `name` when calling `kuzu_client` or Redis functions, and by `agent_type_id` (UUID) when calling Postgres tables that have a foreign key to `agent_types.id`.
- **Idempotent schema init:** `initialize_schema()` from T0.6 uses `IF NOT EXISTS` for all DDL, so calling it again (e.g. during a re-certification in Phase 3) is always safe.
- **Confidence starts at 0.0:** Root concept nodes are created with `confidence=0.0` because they represent *topics to explore*, not *verified facts*. The gap detector (T1.2) will immediately flag these as `unexplored` -- this is intentional and is the very first gap the Socratic loop will surface.
