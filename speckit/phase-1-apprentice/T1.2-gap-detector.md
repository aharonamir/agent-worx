```python
# src/apprentice/gap_detector.py

from datetime import datetime
from src.core.models import KnowledgeGap, GapDetectorResult
from src.infra.kuzu_client import execute


def detect_gaps(agent_type: str) -> GapDetectorResult:
    """
    Query the Kuzu domain graph for the three gap signal types.
    Returns gaps ranked by priority_score = severity × breadth, descending.
    """
    orphans    = _find_orphan_nodes(agent_type)
    low_conf   = _find_low_confidence_edges(agent_type)
    unexplored = _find_unexplored_branches(agent_type)

    all_gaps = orphans + low_conf + unexplored
    all_gaps.sort(key=lambda g: g.priority_score, reverse=True)

    return GapDetectorResult(
        agent_type_id=agent_type,
        gaps=all_gaps,
        total_nodes=_count_nodes(agent_type),
        orphan_count=len(orphans),
        low_confidence_count=len(low_conf),
        unexplored_count=len(unexplored),
    )


def _find_orphan_nodes(agent_type: str) -> list[KnowledgeGap]:
    """Concepts with no edges in either direction."""
    df = execute(agent_type, """
        MATCH (c:Concept)
        WHERE NOT EXISTS { MATCH (c)-[:RELATES_TO]-() }
        RETURN c.id AS id, c.label AS label, c.namespace AS namespace
    """).get_as_df()

    gaps = []
    for _, row in df.iterrows():
        breadth = _count_dependent_concepts(agent_type, row["id"])
        gaps.append(KnowledgeGap(
            concept_id=row["id"],
            concept_label=row["label"],
            gap_type="orphan",
            severity=1.0,
            breadth=breadth,
            priority_score=1.0 * breadth,
            namespace=row["namespace"],
        ))
    return gaps


def _find_low_confidence_edges(agent_type: str,
                                threshold: float = 0.4) -> list[KnowledgeGap]:
    """Source concepts of edges with confidence below threshold."""
    df = execute(agent_type, """
        MATCH (a:Concept)-[r:RELATES_TO]->(b:Concept)
        WHERE r.confidence < $threshold
        RETURN a.id AS id, a.label AS label, a.namespace AS namespace,
               MIN(r.confidence) AS min_confidence
        ORDER BY min_confidence ASC
    """, {"threshold": threshold}).get_as_df()

    gaps = []
    for _, row in df.iterrows():
        severity = 1.0 - float(row["min_confidence"])  # 0.4 conf → 0.6 severity
        breadth  = _count_dependent_concepts(agent_type, row["id"])
        gaps.append(KnowledgeGap(
            concept_id=row["id"],
            concept_label=row["label"],
            gap_type="low_confidence",
            severity=round(severity, 3),
            breadth=breadth,
            priority_score=round(severity * breadth, 3),
            namespace=row["namespace"],
        ))
    return gaps


def _find_unexplored_branches(agent_type: str) -> list[KnowledgeGap]:
    """
    Topic list root nodes (namespace='task', created at seed time)
    that have fewer than 2 outgoing edges — meaning they've barely been explored.
    """
    df = execute(agent_type, """
        MATCH (c:Concept)
        WHERE c.namespace = 'task'
        OPTIONAL MATCH (c)-[r:RELATES_TO]->()
        WITH c, COUNT(r) AS edge_count
        WHERE edge_count < 2
        RETURN c.id AS id, c.label AS label, edge_count
    """).get_as_df()

    gaps = []
    for _, row in df.iterrows():
        edge_count = int(row["edge_count"])
        severity = 0.5 + (0.5 * (1 - edge_count / 2))  # 0 edges → 1.0, 1 edge → 0.75
        breadth  = _count_dependent_concepts(agent_type, row["id"])
        gaps.append(KnowledgeGap(
            concept_id=row["id"],
            concept_label=row["label"],
            gap_type="unexplored",
            severity=round(severity, 3),
            breadth=max(breadth, 1),
            priority_score=round(severity * max(breadth, 1), 3),
            namespace="task",
        ))
    return gaps


def _count_dependent_concepts(agent_type: str, concept_id: str) -> int:
    """Count concepts reachable from this concept (breadth proxy)."""
    df = execute(agent_type, """
        MATCH (a:Concept {id: $id})-[:RELATES_TO*1..3]->(b:Concept)
        RETURN COUNT(DISTINCT b) AS cnt
    """, {"id": concept_id}).get_as_df()
    return int(df["cnt"].iloc[0]) if len(df) > 0 else 0


def _count_nodes(agent_type: str) -> int:
    df = execute(agent_type,
        "MATCH (c:Concept) RETURN COUNT(c) AS cnt"
    ).get_as_df()
    return int(df["cnt"].iloc[0])
```

## Acceptance Criteria

1. Given a graph with 3 nodes and no edges → `orphan_count == 3`, all gaps have `gap_type="orphan"`
2. Given a graph with one edge at confidence 0.2 → `low_confidence_count >= 1`
3. Top-ranked gap is the concept with the highest `priority_score` (severity × breadth)
4. `detect_gaps` on an empty graph returns empty gaps list without raising

```python
# tests/phase_1/test_gap_detector.py

import pytest
from src.apprentice.gap_detector import detect_gaps
from src.infra.kuzu_client import initialize_schema, execute


@pytest.fixture
def agent_type(tmp_path, monkeypatch):
    monkeypatch.setenv("KUZU_GRAPHS_DIR", str(tmp_path))
    atype = "test-gaps"
    initialize_schema(atype)
    return atype


def _add_concept(agent_type, cid, label, namespace="task"):
    execute(agent_type,
        f"CREATE (:Concept {{id:'{cid}', label:'{label}', namespace:'{namespace}', confidence:0.9, created_at:'2025-01-01'}})"
    )


def test_detects_orphan_nodes(agent_type):
    _add_concept(agent_type, "c1", "burger")
    _add_concept(agent_type, "c2", "fries")
    result = detect_gaps(agent_type)
    assert result.orphan_count == 2
    assert all(g.gap_type == "orphan" for g in result.gaps)


def test_orphan_ranks_higher_when_more_dependents(agent_type):
    # c1 has a dependent (c2), c3 is isolated
    _add_concept(agent_type, "c1", "menu")
    _add_concept(agent_type, "c2", "item")
    _add_concept(agent_type, "c3", "isolated")
    execute(agent_type,
        "MATCH (a:Concept{id:'c1'}),(b:Concept{id:'c2'}) CREATE (a)-[:RELATES_TO{relation:'has',confidence:0.9,version:1}]->(b)"
    )
    result = detect_gaps(agent_type)
    # c3 is an orphan with breadth 0; c1 may have low outgoing but higher breadth
    orphans = [g for g in result.gaps if g.gap_type == "orphan"]
    assert len(orphans) >= 1


def test_detects_low_confidence_edge(agent_type):
    _add_concept(agent_type, "c1", "order")
    _add_concept(agent_type, "c2", "payment")
    execute(agent_type,
        "MATCH (a:Concept{id:'c1'}),(b:Concept{id:'c2'}) CREATE (a)-[:RELATES_TO{relation:'triggers',confidence:0.2,version:1}]->(b)"
    )
    result = detect_gaps(agent_type)
    assert result.low_confidence_count >= 1


def test_empty_graph_returns_no_gaps(agent_type):
    result = detect_gaps(agent_type)
    assert result.gaps == []
    assert result.total_nodes == 0


def test_gaps_sorted_by_priority_descending(agent_type):
    for i in range(5):
        _add_concept(agent_type, f"c{i}", f"concept-{i}")
    result = detect_gaps(agent_type)
    scores = [g.priority_score for g in result.gaps]
    assert scores == sorted(scores, reverse=True)
```
